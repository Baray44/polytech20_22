
clc, clear
disp ('Программа моделирования случайно величины и вычисление ее характеристик')
                           %Задание исходных данных
disp('Задайте математическое ожидание и дисперсию для моделирования случайной величины, распределенной по нормальному закону')
mo = input(' mo='); sigm = input(' sigm= ')';

disp('Задайте количество реализаций (100-2000)'), n = input(' n = ');

                           % Моделирование
x = normrnd(mo,sigm,1,n);
                           % Оценка параметров распределения
disp('Вычисленные МО и дисперсия случайной величины:')
mo_= mean(x); sigma_=std(x); pause
                           % Построение графиков
subplot(211), plot(x, 'r'), grid, hold on,
    mo1=mo_*ones(1,n); plot(mo1, 'g'), hold off
    title('Последовательность случайных чисел')
subplot(212), histogram(x,100), grid,
    title('Гистограмма плотности распределения случайной величины'), pause
